## Usage

```tsx copy
import { Theme, ThemeProps } from '@creation-ui/react'

export const App = () => {
  const myTheme: ThemeProps = {
    // ... your theme options
  }

  return (
    <Theme theme={myTheme}>
      <>{/* Rest of your code  */}</>
    </Theme>
  )
}
```
