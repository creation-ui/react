```tsx copy
export const InputExample = (props: Omit<InputProps, 'onChange' | 'ref'>) => {
  const [value, setValue] = useState('')

  return (
    <Input onChange={e => setValue(e.target.value)} value={value} {...props} />
  )
}
```

As you may observe in this example the longer the content inside the Input component
the longer it gets. To fix that you would need to provide your own **width** constraints
either by a **custom wrapper** or passing class to `className` e.g.

```tsx copy
//example with tailwindcss custom width
<Input className='w-[200px]' />
```
