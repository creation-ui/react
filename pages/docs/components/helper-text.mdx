# HelperText

Used in inputs to display helper text or an error message. When content is falsy then it won't be displayed at all.

## Usage

```tsx copy
import { HelperText } from '@creation-ui/react'

;<HelperText
  size={size}
  helperText={error || helperText}
  error={Boolean(error)}
/>
```

import { HelperTextExample, properties } from '@components/examples/helper-text'
import { Playground } from '@components/playground'

<Playground
  config={{
    error: true,
    size: true,
    helperText: true,
  }}
>
