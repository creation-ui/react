  ToggleGroupExample,
  properties,
  optionProps,
} from '@examples/toggle-group'
import { Container } from '@components/container'

import PropsTable from '@components/props-table'

```tsx copy
import { ToggleGroup, ToggleGroupProps } from '@creation-ui/react'

export const ToggleGroupExample = ({ size }: ToggleGroupProps) => {
  const options = [
    { value: '1', label: <Icon icon='3d_rotation' /> },
    { value: '2', label: <Icon icon='access_alarms' /> },
    { value: '3', label: <Icon icon='add_to_queue' /> },
  ]

  return <ToggleGroup size={size} options={options} label='Select option' />
}
```

<Container>
  <ToggleGroupExample label='Radio option' />
</Container>

## Sizes

This component supports `size` prop with values `sm`, `md`, `lg`:

  variants={ELEMENT_SIZES}
