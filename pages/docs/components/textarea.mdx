# TextArea

import { TextAreaExample, properties } from '@examples/textarea'
import { Container } from '@components/container'
import { UnfoldVariants } from '@components/unfold-variants'
import { ELEMENT_SIZES, ELEMENT_STATUS } from '@creation-ui/core'
import PropsTable from '@components/props-table'

```tsx copy
export const TextAreaExample = (props: Omit<TextAreaProps, 'onChange' | 'ref'>) => {
  const [value, setValue] = useState('')

  return (
    <TextArea
      //
      onChange={e => setValue(e.target.value)}
      value={value}
      {...props}
    />
  )
}
```

<Container>
  <TextAreaExample label='TextArea' debug />
</Container>

As you may observe in this example the longer the content inside the TextArea component
the longer it gets. To fix that you would need to provide your own **width** constraints
either by a **custom wrapper** or passing class to `className` e.g.

```tsx copy
//example with tailwindcss custom width
<TextArea className='w-[200px]' />
```
