# Tooltip

<Callout>Component is visible after hovering over the trigger element.</Callout>

```tsx copy
import { Button, Tooltip } from '@creation-ui/react'

export const TooltipExample = ({ ...props }: TooltipProps) => {
  return (
    <Tooltip {...props} content='This is important button' position='bottom'>
      <Button variant='contained'>Submit</Button>
    </Tooltip>
  )
}
```

<Container>
  <TooltipExample />
</Container>

## Position

This component supports the `position` prop with values `left`, `top`, `bottom`, `right`.
