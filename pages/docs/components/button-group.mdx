# Button Group

```tsx copy
import { Button, ButtonProps, ButtonGroup } from '@creation-ui/react'

interface ButtonGroupExampleProps {
  options?: string[]
  variant?: ButtonProps['variant']
}

export const ButtonGroupExample = ({
  options,
  variant,
}: ButtonGroupExampleProps) => {
  return (
    <ButtonGroup>
      {options?.map(option => (
        <Button variant={variant}>{option}</Button>
      ))}
    </ButtonGroup>
  )
}
```

`div` wrapper for buttons.

<Container>
  <ButtonGroupExample options={[1, 2, 3]} />
</Container>
