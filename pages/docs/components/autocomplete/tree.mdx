## Displaying tree structures

If you want to display tree structure to select from e.g. nested categories then you need to build tree from your data and provide children array to each node.
If data will have children array, then it will be displayed as tree structure. You can enable this feature by setting `isTree` prop to `true`.

```tsx copy
import { Autocomplete } from '@creation-ui/react/components'
import { buildTree } from '@creation-ui/react/components/tree/utils'
import categories from '../tree-data.json'
import { useState } from 'react'

type Category = {
  //
  id: number
  name: string // or change to label and dispose of `getOptionLabel`
  parent_id?: number | null
  children?: Category[]
}

type SelectorType = Category | undefined | null

export const AutocompleteTreeExample = () => {
  const tree = buildTree(categories)
  const [value, setValue] = useState<SelectorType>(null)

  const onChange = (value: SelectorType) => {
    setValue(value)
  }

  return (
    <Autocomplete
      label={'Tree select'}
      clearable
      getOptionLabel={({ name }: Category) => name}
      tree
      onChange={onChange}
      value={value}
      options={tree}
    />
  )
}
```

</Container>
