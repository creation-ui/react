# ProgressBar

```tsx copy
import { ProgressBar } from '@creation-ui/react'

export const ProgressBarExample = ({ ...props }: ProgressBarProps) => {
  return <ProgressBar {...props} />
}
```

<Container>
  <ProgressBarExample value={50} />
</Container>

## Size

This component supports the `size` prop with values `sm`, `md`, `lg`.

<UnfoldVariants
  component={ProgressBarExample}
  variants={[...ELEMENT_SIZES]}
  prop='size'
  nameProp='key'
  componentProps={{
    value: 50,
  }}
/>

## Showing value

Value can be displayed passing `showValue` prop

```tsx copy
<ProgressBar value={65} showValue />
```

<Container>
  <ProgressBar value={65} showValue />
</Container>

## Formatting value

Value can be custom formatted with your own function that takes `value` and returns `ReactNode`.
`showValue` needs to be set to true for this to work.

```tsx copy
<ProgressBar
  value={65}
  formatDisplayValue={value => `Preparing for lunch: ${value}%`}
  showValue
/>
```

<Container>
  <ProgressBarExample
    value={65}
    formatDisplayValue={value => `Preparing for lunch: ${value}%`}
    showValue
  />
</Container>
