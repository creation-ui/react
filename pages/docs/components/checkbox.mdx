# Checkbox

import { CheckboxExample } from '@examples/checkbox'
import { Container } from '@components/container'
import { UnfoldVariants } from '@components/unfold-variants'
import { ELEMENT_SIZES, ELEMENT_STATUS } from '@creation-ui/core'
import PropsTable from '@components/props-table'
import { properties } from '@examples/checkbox'

```tsx copy
import { Checkbox, CheckboxProps } from '@creation-ui/Checkbox'

export const CheckboxComponet = ({
  ...props
}: Omit<CheckboxProps, 'onChange'>) => {
  const [checked, setChecked] = useState<boolean>(false)

  return <Checkbox onChange={setChecked} checked={checked} {...props} />
}
```

</Container>
