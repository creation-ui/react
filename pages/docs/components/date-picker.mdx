# DatePicker

DatePicker component just styling around [React Datepicker package](https://reactdatepicker.com/).
Please refer to the [docs](https://reactdatepicker.com/) on how to use it

```tsx copy
import { DatePicker, DatePickerProps } from '@creation-ui/date-picker'

export const DatePickerExample = ({ ...props }: DatePickerProps) => {
  const [startDate, setStartDate] = useState<Date | null>(new Date())
  return (
    <div className='flex flex-col gap-5'>
      <pre className='text-xs'>{JSON.stringify(startDate)}</pre>
      <DatePicker
        {...props}
        selected={startDate}
        onChange={date => setStartDate(date)}
      />
    </div>
  )
}
```

</Container>
